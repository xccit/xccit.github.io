vite+vue3封装request.js

    npm init vite@latest

    选择vue

初始化工程

1.安装axios

npm install axios

2.封装request.js

在src下新建utils文件夹，新建request.js

import axios from "axios"
import { ElMessage } from 'element-plus';
import {useTokenStore} from "@/store/token.js";

const baseURL = '/api'
// 创建axios实例
const service = axios.create({
    baseURL: baseURL, // 设置默认的 API 地址，url = baseURL + url（使用proxy代理时此处可不写）
    timeout: 5000,	// 设置请求超时时间
})

//请求拦截器
service.interceptors.request.use(
    (config)=>{
        let tokenStore = useTokenStore()
        if (tokenStore.token){
            config.headers.Authorization = tokenStore.token
        }
        return config
    },
    (error)=>{
        ElMessage.error(error.message)
        return Promise.reject(error)
    }
)

// 响应拦截器
service.interceptors.response.use(
    // 响应数据，2xx 范围内的状态码都会触发该函数。
    (response) => {
        if (response.data.code == 200){
            const { status, data } = response;
            if (status === 200) {
                const { code, message } = data;
                // 根据后端返回的自定义状态码 code 进行错误信息提示（根据具体需求确定是否需要书写）
                switch (code) {
                    case 401:
                        ElMessage({ message: message, type: 'error' })
                        return Promise.reject(data);
                    default:
                        return data;
                }
            }
        }else{
            return Promise.reject(response.data)
        }
    },
    (error) => {
        if (error.response) {
            const { status, data } = error.response;
            ElMessage({ message: data.message, type: 'error' })
            // if (status === 500) {
            //     ElMessage({ message: data.message, type: 'error' })
            // } else if (status === 401) {
            //     ElMessage({ message: data.message, type: 'error' })
            // } else {
            //     ElMessage({ message: data.message || '未知错误！', type: 'error' })
            // }
        } else if (error.request) {
            // 请求已发送，未收到响应
            ElMessage({ message: error.message || '请求已发送，未收到响应信息！', type: 'error' })
        } else {
            // 其他错误
            ElMessage({ message: error.message, type: 'error' })
        }
        // 若简写，上面代码可省略，只写此一行
        return Promise.reject(error);
    }
)

export default service;

3.使用pinia进行状态管理

后端接口一般都是需要验证token来进行访问的：

新建src/store/token.js

import {defineStore} from 'pinia'
import {ref} from "vue";

export const useTokenStore = defineStore('token',()=>{
    const token = ref('')
    const setToken = (newToken)=>{
        token.value = newToken
    }

    const removeToken = ()=>{
        token.value = ''
    }

    return {token,setToken,removeToken}
})

这里定义了pinia状态管理的js文件，但是我们需要在main.js中使用pinia

import {createPinia} from 'pinia'
const pinia = createPinia()

const app = createApp(App)
app.use(ElementPlus,{locale})
    .use(router)
    .use(pinia)
app.mount('#app')

忽略掉ElementPlus与locale以及router,并且在上面的request.js文件中，同样引入了token.js,这是为了方便将登录后获得的token传给请求拦截器，在请求头Headers中加入token

测试

定义接口文件api/user.js

import request from '@/utils/request.js'

const baseUrl = '/user'
export const register = (params)=>{
    return request({
        method: "POST",
        url: `${baseUrl}/register`,
        params: params
    })
}

export const login = (params)=>{
    return request({
        method: "POST",
        url: `${baseUrl}/login`,
        params: params
    })
}

export const logout = ()=>{
    return request({
        method: "POST",
        url: `${baseUrl}/logout`
    })
}

定义接口文件category.js

import request from '@/utils/request.js'

const baseUrl = '/category'
export const list = ()=>{
    return request({
        method: 'GET',
        url: `${baseUrl}`
    })
}

export const insert = (category)=>{
    return request({
        method: 'POST',
        url:`${baseUrl}`,
        data: category
    })
}

export const update = (category)=>{
    return request({
        method: 'PUT',
        url: `${baseUrl}`,
        data: category
    })
}

export const remove = (id)=>{
    return request({
        method: 'DELETE',
        url: `${baseUrl}/${id}`
    })
}

export const detail = (id)=>{
    return request({
        method: 'GET',
        url: `${baseUrl}/detail`,
        params: id
    })
}
